    //Import and Export
            1. Default export
Use this when a file exports one main thing. You can name the import anything you want.

export default function Button() {
  return <button>Click me</button>;
}

or 
 function Button() {
  return <button>Click me</button>;
}

export default Button;

        * TO import default export

import CustomButton from './Button';  or import Button from './Button';

function App() {
  return <CustomButton />;
}

            2. Named export
Use this when a file exports multiple things. You must use curly braces {} and match the exact export names.

export function HomeIcon() { return <span>🏠</span>; }
export function UserIcon() { return <span>👤</span>; }

        * TO import named export

import { HomeIcon, UserIcon } from './Icons';   // must use curly braces and match the exact export identifiers(names)

function App() {
  return <div><HomeIcon /><UserIcon /></div>;
}

        3. Combined Export (Default+Named)

// Default export
export default function Alert({ children }) {
  return <div className="alert">{children}</div>;
}

// Named exports
export const SEVERITY_HIGH = 'high';
export const SEVERITY_LOW = 'low';


        * TO import Combined export (Default+Named)

import Alert, { SEVERITY_HIGH, SEVERITY_LOW } from './Alert';

function App() {
  return (
    <Alert severity={SEVERITY_HIGH}>
      Critical system error!
    </Alert>
  );
}


                    **Difference between backend and  Frontend Import/Export 

The difference comes down to two different module systems created at different times in JavaScript's history.
->The Backend Version (require / module.exports) is called CommonJS (CJS).
->The Frontend Version (import / export) is called ECMAScript Modules (ESM). 

** Why Frontend Uses import and exportIn 2015, the official JavaScript standard committee released a new version of JavaScript (ES6) and introduced a native way to handle modules, called ECMAScript Modules (ESM).Because it is the official standard, modern frontend frameworks (like React, Vite, and Next.js) adopted it completely.


** Can you use import/export in the backend now?
Yes. Modern versions of Node.js fully support the import/export (ESM) syntax. 
You can use it in your backend by doing one of two things:
Change your file extension from .js to .mjs.
Add "type": "module" inside your backend's package.json file.

{
  "name": "my-backend-app",
  "version": "1.0.0",
  "description": "Node backend using ES Modules",
  "main": "server.js",
  "type": "module", // add this 
  "scripts": {
    "start": "node server.js"
  },
  "dependencies": {
    "express": "^4.19.2"
  }
}
